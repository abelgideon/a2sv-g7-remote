class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class Stack:
    def __init__(self):
        self.head = None
        self.size = 0

    def push(self, data):
        if self.head == None:
            self.head = Node(data)
        else:
            newNode = Node(data)
            newNode.next = self.head
            self.head = newNode
        self.size += 1
    
    def pop(self):
        if self.head == None:
            return None
        else:
            popped = self.head
            self.head = self.head.next
            popped.next = None
            self.size -= 1
            return popped.data
    
    def peek(self):
        if self.head == None:
            return None
        return self.head.data

class Solution:
    def isValid(self, s: str) -> bool:
        stack = Stack()

        pairs = {
            ")": "(",
            "}": "{",
            "]": "["
        }
        opening = set("({[")

        if s[0] not in opening:
            return False

        for ch in s:
            if ch in opening:
                stack.push(ch)
            else:
                top = stack.peek()
                if pairs[ch] == top:
                    stack.pop()
                else:
                    return False
        
        return True if stack.size == 0 else False
