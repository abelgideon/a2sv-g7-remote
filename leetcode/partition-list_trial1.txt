# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def insertNode(self, head, val):
        newNode = ListNode(val)
        curr = head
        while curr.next:
            curr = curr.next
        
        curr.next = newNode

    def partition(self, head: Optional[ListNode], x: int) -> Optional[ListNode]:
        res = ListNode(0, None)

        curr = head
        while curr:
            if curr.val < x:
                self.insertNode(res, curr.val)
            curr = curr.next
        
        curr = head
        while curr:
            if curr.val >= x:
                self.insertNode(res, curr.val)
            curr = curr.next
        
        return res.next
