class Solution:
    def predictTheWinner(self, nums: List[int]) -> bool:
        def p1Score(left, right):
            if left == right:
                return nums[left]
            
            pickLeft = nums[left] - p1Score(left + 1, right)
            pickRight = nums[right] - p1Score(left, right - 1)

            return max(pickLeft, pickRight)
        
        if p1Score(0, len(nums)-1) >= 0:
            return True
        
        return False
