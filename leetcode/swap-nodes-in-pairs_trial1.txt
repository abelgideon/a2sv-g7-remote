# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head or not head.next:
            return head

        dummy = ListNode(0, head)
        prev = dummy
        curr = head
        nxt = curr.next

        while curr and nxt:
            tmp = nxt.next
            nxt.next = curr
            curr.next = tmp
            prev.next = nxt

            prev = curr
            curr = curr.next
            if curr:
                nxt = curr.next
        
        return dummy.next
