class Solution:
    def countVowelSubstrings(self, word: str) -> int:
        count = 0
        vowels = set("aeiou")

        for i in range(len(word)):
            currCount = set()
            for j in range(i, len(word)):
                currCount.add(word[j])

                if currCount == vowels:
                    count += 1

        return count
