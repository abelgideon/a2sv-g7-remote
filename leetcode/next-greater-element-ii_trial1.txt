class Solution:
    def nextGreaterElements(self, nums: List[int]) -> List[int]:
        stack = []
        N = len(nums)

        nums += nums
        res = [-1] * len(nums)

        for i in range(len(nums)):
            while stack and nums[i] > nums[stack[-1]]:
                res[stack.pop()] = nums[i]

            stack.append(i)
        
        return res[:N]
