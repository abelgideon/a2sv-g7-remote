class Solution:
    def maxArea(self, height: List[int]) -> int:
        L = 0
        R = len(height) - 1

        maxWater = float("-inf")

        while L < R:
            currWater = min(height[L], height[R]) * (R - L)

            maxWater = max(maxWater, currWater)

            if height[L] < height[R]:
                L += 1
            else:
                R -= 1
        
        return maxWater
