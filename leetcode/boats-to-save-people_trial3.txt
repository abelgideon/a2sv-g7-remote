class Solution:
    def numRescueBoats(self, people: List[int], limit: int) -> int:
        people.sort()
        print(people)
        boatCount = 0

        L = 0
        R = len(people) - 1

        while L <= R:
            if people[L] + people[R] > limit:
                boatCount += 1
                R -= 1
            else:
                boatCount += 1
                L += 1
                R -= 1
        
        return boatCount
