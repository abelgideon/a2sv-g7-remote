class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        L = 0
        window = defaultdict(int)
        maxFreq = 0
        res = 0

        for R in range(len(s)):
            window[s[R]] += 1

            maxFreq = max(window[s[R]], maxFreq)

            while (R - L + 1) - maxFreq > k:
                window[s[L]] -= 1
                L += 1
            
            res = max(res, R - L + 1)
        
        return res
