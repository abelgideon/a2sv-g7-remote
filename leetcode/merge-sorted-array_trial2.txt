class Solution:
    def merge(self, nums1: List[int], m: int, nums2: List[int], n: int) -> None:
        """
        Do not return anything, modify nums1 in-place instead.
        """
        T = m - 1
        B = n - 1
        K = len(nums1) - 1

        while B >= 0:
            if T >= 0 and nums1[T] >= nums2[B]:
                nums1[K] = nums1[T]
                T -= 1
            else:
                nums1[K] = nums2[B]
                B -= 1
            
            K -= 1
