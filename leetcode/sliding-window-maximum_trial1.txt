class Solution:
    def maxSlidingWindow(self, nums: List[int], k: int) -> List[int]:
        maxQ = deque()
        res = []

        for i in range(len(nums)):
            if maxQ and maxQ[0] <= i - k:
                maxQ.popleft()
            
            while maxQ and nums[maxQ[-1]] < nums[i]:
                maxQ.pop()
            
            maxQ.append(i)

            if i >= k - 1:
                res.append(nums[maxQ[0]])

        return res
