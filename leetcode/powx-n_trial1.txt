class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0: return 1
        if x == 0: return 0

        isNeg = True if n < 0 else False
        n = abs(n)

        res = 1
        while n > 0:
            if n % 2 == 1:
                res *= x

            x *= x
            n //= 2
        
        return 1/res if isNeg else res
