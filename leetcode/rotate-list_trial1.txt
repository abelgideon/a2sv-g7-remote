# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or not head.next:
            return head

        curr = head
        length = 1

        while curr.next:
            length += 1
            curr = curr.next
        
        curr.next = head
        k %= length

        prev = None
        for _ in range(length - k):
            prev = head
            head = head.next
        
        prev.next = None

        return head
